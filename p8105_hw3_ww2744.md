``` r
data("ny_noaa")
ny_noaa = 
  ny_noaa |>
  janitor::clean_names() |>
  separate(col = date, into = c("year", "month", "day"), sep = "-", , convert = TRUE) |>
  mutate(
    tmax = as.numeric(tmax) / 10,
    tmin = as.numeric(tmin) / 10,
  )

ny_noaa |>
  summarize(
    most_common_snow = names(which.max(table(na.omit(snow))))
  )
```
